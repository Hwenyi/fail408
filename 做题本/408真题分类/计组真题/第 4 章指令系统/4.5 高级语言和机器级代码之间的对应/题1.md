```cpp
int f1(unsigned n)
1   00401020 55         push ebp
    ...      ...        ...
        for (unsigned i = 0; i <= n-1; i++)
    ...      ...        ...
20  0040105E 39 4D F4   cmp dword ptr [ebp-0Ch],ecx
    ...      ...        ...
        { power *= 2;
    ...      ...        ...
23  00401066 D1 E2      shl edx,1
    ...      ...        ...
        return sum;
```
